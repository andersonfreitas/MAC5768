import time

lena = imread('../images/images256/lena.png')
gray()
t1 = time.time()
lena_cw = zip(*lena[::-1])
# lena_ccw = zip(*lena)[::-1]
lena_ccw = zeros(shape(lena)) #[[0 for _ in range(len(lena))] for _ in range(len(lena[0]))]

h = len(lena)
w = len(lena[0])

for i in range(h):
    for j in range(w):
        lena_ccw[w - 1 - j][i] = lena[i][j]

t2 = time.time()
# pylab.rcParams['figure.figsize'] = (15.0, 8.0)

subplot(1,2,1)
imshow(lena)

subplot(1,2,2)
imshow(lena_ccw)

print "Tempo total =", t2-t1

Tempo total = 0.19414806366

Com esta mesma imagem, calcule e apresente a mdia e o desvio padro usando loops (range). Em seguida, calcule e apresente usando as funes do numpy. Existe alguma diferena?

Mdia: $$ \mu =\frac { 1 }{ H } \sum _{ \upsilon = 0 }^{ { \upsilon }_{ max } } \left( \upsilon { H }_{ \upsilon } \right) $$

Desvio padro: $$ \sigma =\sqrt { \frac { \sum _{ \upsilon =0 }^{ { \upsilon }_{ max } } \left[ { \left( \upsilon -\mu \right) }^{ 2 }H _{ \upsilon } \right] }{ N - 1 } } $$
