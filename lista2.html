Apresente (graficamente) os histogramas usando a funo plot do mdulo matplotlib. Compare o resultado da sua implementao com a da toolbox ia636.

from skimage.io import imread
import ia636

pylab.rcParams['figure.figsize'] = (8.0, 15.0)

def histograma(img):
    histo = [0]*256
    
    for i in range(shape(img)[0]):
        for j in range(shape(img)[1]):
            histo[img[i][j]] += 1
    return histo

img1 = imread('../images/images256/lena.png')

def histograma_rel(img):
    histo = [0.0] * 256
    
    for i in range(shape(img)[0]):
        for j in range(shape(img)[1]):
            histo[img[i][j]] += 1
    
    N = size(img)

    for v in range(len(histo)):
        histo[v] = histo[v] / N

    return histo

img1 = imread('../images/images256/lena.png')
hist1a = histograma(img1)
hist1b = histograma_rel(img1)

pylab.rcParams['figure.figsize'] = (15.0, 5.0)
subplot(2,2,1)
plot(hist1a)
subplot(2,2,2)
plot(hist1b)
